14"""The unique worker functionality for this service is contained here. 

15 

16The entry-point is the **cb_subscription_handler** 

17 

18The design and flow leverage a few constraints that are placed upon it 

19by NATS Streaming and using AWAIT on the default loop. 

20- NATS streaming queues require one message to be processed at a time. 

21- AWAIT on the default loop effectively runs synchronously 

22 

23If these constraints change, the use of Flask-SQLAlchemy would need to change. 

24Flask-SQLAlchemy currently allows the base model to be changed, or reworking 

25the model to a standalone SQLAlchemy usage with an async engine would need 

26to be pursued. 

27"""

163async def cb_subscription_handler(msg: nats.aio.client.Msg): 

164 """Use Callback to process Queue Msg objects.""" 

165 try: 

166 logger.info('Received raw message seq:%s, data= %s', msg.sequence, msg.data.decode()) 

167 payment_token = extract_payment_token(msg) 

168 logger.debug('Extracted payment token: %s', payment_token) 

169 await process_payment(payment_token, FLASK_APP) 

170 except OperationalError as err: 

171 logger.error('Queue Blocked - Database Issue: %s', json.dumps(payment_token), exc_info=True) 

172 raise err # We don't want to handle the error, as a DB down would drain the queue 

173 except FilingException: 

174 # log to sentry and absorb the error, ie: do NOT raise it, otherwise the message would be put back on the queue 

175 if APP_CONFIG.ENVIRONMENT == 'prod': 

176 capture_message('Queue Error: cannot find filing: %s' % json.dumps(payment_token), level='error') 

177 logger.error('Queue Error - cannot find filing: %s', json.dumps(payment_token), exc_info=True) 

178 except (QueueException, Exception): # pylint: disable=broad-except # noqa B902 

179 # Catch Exception so that any error is still caught and the message is removed from the queue 

180 capture_message('Queue Error:' + json.dumps(payment_token), level='error') 

181 logger.error('Queue Error: %s', json.dumps(payment_token), exc_info=True)
